% Define a custom \centerdot marker
\newcommand{\mycenterdot}{\hspace{1em}$\cdot$\hspace{1em}}

% Define a custom function for creating a list with custom marker
\newcommand{\myitemlist}[1]{%
    \begin{itemize}
        \renewcommand{\labelitemi}{\mycenterdot} % Set the marker to custom \centerdot
        #1 % Content of the item list
    \end{itemize}%
}
